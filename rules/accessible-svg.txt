import type { IMessageTranslation, ISkipReason } from '../../../../interfaces/rule-issue.interface';
import { $runnerSettings } from '../../../../constants/auditor';
import {
  $severity, $auditRules, AuditStandards, $auditRuleNodeSkipReason
} from '../../../../constants/audits';
import { AbstractRule } from '../../../abstract-rule';
import { type IAbstractRuleConfig } from '../../../../interfaces/rule.interface';

import { Config } from '../../../../config';
import { DomUtility } from '../../../../utils/dom';
import { LevelType } from '../../../../constants/levelType';
import { ObjectUtility } from '../../../../utils/object';
import { TextUtility } from '../../../../utils/text';

/*
 * https://www.w3.org/TR/svg-aam-1.0/
 *
 * SVG user agents MUST provide an accessible object in the accessibility tree for rendered SVG elements that meet any of the following criteria, unless they are excluded from the accessibility tree per the rules in Excluding Elements from the Accessibility Tree:
 *
 * It has at least one direct child ‘title’ element or ‘desc' element that is not empty after trimming whitespace. User agents MAY include elements with these child elements without checking for valid text content.
 * It has a non-empty (after trimming whitespace) ‘aria-label’ attribute or ‘aria-roledescription’ attribute.
 * It has an ‘aria-labelledby’ attribute or ‘aria-describedby’ attribute containing valid IDREF tokens. User agents MAY include elements with these attributes without checking for validity.
 * It has a valid integer 'tabindex' attribute.
 * The author has provided an allowed, non-abstract WAI-ARIA role other than none or presentation.
 */

const suggestedSolutions: string = `${TextUtility.escape(
  '<title>, <desc>, <text>, aria-label, aria-labelledby, aria-describedby, aria-roledescription'
)}`;

export class AccessibleSvg extends AbstractRule {
  protected override selector = 'svg:not([role="presentation"])';
  protected ruleConfig: IAbstractRuleConfig = {
    disabilitiesImpacted: [],
    level: LevelType.A,
    ruleId: $auditRules.accessible_svg,
    severity: $severity.high,
    standard: AuditStandards.wcag,
    standardMetaData: null
  };

  private elementShouldBeSkipped(element: HTMLElement | SVGElement, isElementVisible: boolean): ISkipReason {
    const skipReason: ISkipReason = {
      elementShouldBeSkipped: true,
      reason: null
    };

    const ariaHidden: Attr | null = DomUtility.getElementAttribute(element, 'aria-hidden');

    if (ariaHidden && ariaHidden.value === 'true') {
      skipReason.reason = $auditRuleNodeSkipReason.hiddenFromAssistiveTechnologies;

      return skipReason;
    }

    if (Config.get($runnerSettings.includeHidden) && isElementVisible === false) {
      skipReason.reason = $auditRuleNodeSkipReason.skipHiddenElements_elementNotVisible;

      return skipReason;
    }

    skipReason.elementShouldBeSkipped = false;

    return skipReason;
  }

  public validate(htmlOrSVGElements: SVGElement[]): void {
    const checkForAnAccessibilityIssues = (svgElement: SVGElement): void => {
      if (DomUtility.isHiddenForAT(svgElement)) {
        return;
      }

      const isElementVisible: boolean = DomUtility.isElementVisible(svgElement);

      if (Config.get($runnerSettings.includeHidden) === false && isElementVisible === false) {
        return;
      }

      const skipReason: ISkipReason = this.elementShouldBeSkipped(svgElement, isElementVisible);

      if (skipReason.elementShouldBeSkipped) {
        this.validator.report({
          messageTranslation: [
            {
              messageTranslationId: `skip_reason_${skipReason.reason}`
            }
          ],
          node: svgElement,
          requiresManualVerification: false,
          ruleId: this.ruleConfig.ruleId,
          skipReason: skipReason.reason
        });

        return;
      }

      const role: string | null = svgElement.getAttribute('role');
      const hasSemanticRole: boolean = !role || ['img', 'graphics-document', 'graphics-symbol'].includes(role);

      // Only consider inherited names for presentational purposes
      if (hasSemanticRole === false) {
        // 1. Bail out if self OR ancestor already has an accessible name
        const inheritedName: string = DomUtility.getAccessibleNameFromSelfOrAncestors(svgElement);

        if (inheritedName.trim().length > 0) {
          return;
        }
      }

      // 2. Collect possible naming sources
      const titleElement: HTMLTitleElement | null = svgElement.querySelector('title');
      const descElement: SVGDescElement | null = svgElement.querySelector('desc');
      const textElement: SVGTextElement | null = svgElement.querySelector('text');

      const ariaLabel: string | null = svgElement.getAttribute('aria-label');
      const ariaLabelledBy: string | null = svgElement.getAttribute('aria-labelledby');
      const ariaRoleDescription: string | null = svgElement.getAttribute('aria-roledescription');
      const ariaDescribedBy: string | null = svgElement.getAttribute('aria-describedby');
      const tabIndex: string | null = svgElement.getAttribute('tabindex');

      const reportMessages: IMessageTranslation[] = [];
      let hasValidName = false;

      // 3. Validate tabindex (never a naming source)
      if (typeof tabIndex === 'string' && ObjectUtility.isNumber(tabIndex) === false) {
        reportMessages.push({
          messageTranslationId: 'accessible_svg_report_message_4',
          messageTranslationParams: [tabIndex]
        });
      }

      // 4. Validate name-providing sources

      const hasNonEmptyText = (el: Element | null): boolean => {
        return Boolean(el?.textContent && el.textContent.trim().length > 0);
      };

      if (hasNonEmptyText(titleElement)) {
        hasValidName = true;
      } else if (titleElement) {
        reportMessages.push({
          messageTranslationId: 'accessible_svg_report_message_2',
          messageTranslationParams: ['title']
        });
      }

      if (hasNonEmptyText(descElement)) {
        hasValidName = true;
      } else if (descElement) {
        reportMessages.push({
          messageTranslationId: 'accessible_svg_report_message_2',
          messageTranslationParams: ['desc']
        });
      }

      if (hasNonEmptyText(textElement)) {
        hasValidName = true;
      } else if (textElement) {
        reportMessages.push({
          messageTranslationId: 'accessible_svg_report_message_2',
          messageTranslationParams: ['text']
        });
      }

      if (typeof ariaLabel === 'string') {
        if (ariaLabel.trim().length > 0) {
          hasValidName = true;
        } else {
          reportMessages.push({
            messageTranslationId: 'accessible_svg_report_message_2',
            messageTranslationParams: ['aria-label']
          });
        }
      }

      if (typeof ariaRoleDescription === 'string') {
        if (ariaRoleDescription.trim().length > 0) {
          hasValidName = true;
        } else {
          reportMessages.push({
            messageTranslationId: 'accessible_svg_report_message_2',
            messageTranslationParams: ['aria-roledescription']
          });
        }
      }

      // 5. aria-labelledby validation (name-providing)
      const getMissingIdRefs = (value: string): string[] => {
        return value.split(/\s+/).filter((id): boolean => {
          return document.getElementById(id) === null;
        });
      };

      const labelledByElements: HTMLElement[] | undefined = (svgElement as any).ariaLabelledByElements as HTMLElement[] | undefined;

      if (Array.isArray(labelledByElements)) {
        if (labelledByElements.length > 0) {
          hasValidName = true;
        } else if (typeof ariaLabelledBy === 'string') {
          // aria-labelledby present but resolved to nothing
          reportMessages.push({
            messageTranslationId: 'accessible_svg_report_message_3',
            messageTranslationParams: ['aria-labelledby', ariaLabelledBy]
          });
        }
      } else if (typeof ariaLabelledBy === 'string') {
        const missingIds: string[] = getMissingIdRefs(ariaLabelledBy);

        if (missingIds.length === 0) {
          hasValidName = true;
        } else {
          reportMessages.push({
            messageTranslationId: 'accessible_svg_report_message_3',
            messageTranslationParams: ['aria-labelledby', missingIds.join(', ')]
          });
        }
      }

      // 6. aria-describedby validation (NOT a name source)
      if (typeof ariaDescribedBy === 'string') {
        const missingIds: string[] = getMissingIdRefs(ariaDescribedBy);

        if (missingIds.length > 0) {
          reportMessages.push({
            messageTranslationId: 'accessible_svg_report_message_3',
            messageTranslationParams: ['aria-describedby', missingIds.join(', ')]
          });
        }
      }

      // 7. Final decision
      if (hasValidName) {
        return;
      }

      if (reportMessages.length === 0) {
        reportMessages.push({
          messageTranslationId: 'accessible_svg_report_message_1',
          messageTranslationParams: [
            svgElement.nodeName.toLowerCase(),
            suggestedSolutions
          ]
        });
      }

      this.validator.report({
        messageTranslation: reportMessages,
        node: svgElement,
        requiresManualVerification: false,
        ruleId: this.ruleConfig.ruleId,
        skipReason: null
      });
    };

    for (const svgElement of htmlOrSVGElements) {
      checkForAnAccessibilityIssues(svgElement);
    }
  }
}
